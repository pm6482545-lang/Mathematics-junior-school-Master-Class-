\section*{1.1 Place Value and Total Value}

\begin{tcolorbox}[colback=lightgrey, colframe=primaryblue, title=\textbf{Concept Summary}, arc=5pt]
    \textbf{Place Value:} Refers to the \textit{name of the position} a digit occupies in a number. \\
    \textbf{Total Value:} Refers to the \textit{numerical worth} of a digit based on its position. \\
    \textbf{Formula:} $\text{Total Value} = \text{Digit} \times \text{Place Value Representation}$
    
    \vspace{10pt}
    \centering
    \small
    \begin{tabularx}{\textwidth}{|X|c|c|c|c|c|c|c|}
    \hline
    \textbf{Digit} & 7 & 4 & 1 & 0 & 3 & 9 & 5 \\ \hline
    \textbf{Place Value} & Millions & H-Th & T-Th & Th & H & T & O \\ \hline
    \end{tabularx}
\end{tcolorbox}

\begin{solvingbox}{Question 1}
Identify the \textbf{place value} and the \textbf{total value} of the underlined digit in the number: $4,\underline{7}82,105$. \\
\textbf{Working:} \\
Underlined digit is 7. \\
Position: 2nd place from the left in the millions period (Hundreds of Thousands). \\
Place Value: \textbf{Hundred Thousands} \hfill $M_1$ \\
Total Value: $7 \times 100,000 = \mathbf{700,000}$ \hfill $A_1$
\end{solvingbox}

\begin{solvingbox}{Question 2}
What is the difference between the total value of digit 8 and digit 3 in the number $843,210$? \\
\textbf{Working:} \\
Total Value of 8: $8 \times 100,000 = 800,000$ \hfill $M_1$ \\
Total Value of 3: $3 \times 1,000 = 3,000$ \hfill $M_1$ \\
Difference: $800,000 - 3,000 = \mathbf{797,000}$ \hfill $A_1$
\end{solvingbox}
